package playground.rost.Controller.UIElements;

import java.awt.BorderLayout;

import javax.swing.JFrame;
import javax.swing.JLabel;

import org.matsim.api.basic.v01.Id;

import playground.dressler.ea_flow.Flow;
import playground.rost.Controller.TimeHandler;

public class FlowAnalyzer extends JFrame implements TimeHandler{

	Flow flow;
	FlowAnalyzeMap mGUI;
	TimePanel timePanel;
	JLabel statusInfo;
	
	
	public FlowAnalyzer(Flow flow, Id superSID)
	{
		super("FlowAnalyzer");
		this.setSize(1000,1000);
        this.setSize(200,344);
        this.setSize(1000,1000);
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        setVisible(false);
        mGUI = new FlowAnalyzeMap(flow, superSID);
        this.setLayout(new BorderLayout());
        this.add(mGUI, BorderLayout.CENTER);
        timePanel = new TimePanel(0);
        timePanel.subscribeTimeEvents(this);
        this.add(timePanel, BorderLayout.NORTH);
        statusInfo = new JLabel();
        this.add(statusInfo, BorderLayout.SOUTH);
        setVisible(true);
		this.flow = flow;
	}
	
	public void timeChanged(int time)
	{
		statusInfo.setText("time: " + time);
		mGUI.setTime(time);
	}
	
	
	
}
